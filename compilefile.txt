# boot/
nasm -f elf32 boot/boot.asm -o boot.o

# helper/
dogec -c helper/dogeio_vbe.c -o dogeio_graphics.o
dogec -c helper/vbe.c -o vbe.o
dogec -c helper/font.c -o font.o
dogec -c helper/dogeio.c -o dogeio.o
dogec -c helper/time.c -o time.o
dogec -c helper/string.c -o string.o
dogec -c helper/ports.c -o ports.o

# kernel/
dogec -c kernel/suchboot.c -o suchboot.o
dogec -c kernel/data.c -o data.o
dogec -c kernel/info.c -o info.o
dogec -c kernel/serial.c -o serial.o
dogec -c kernel/dogeshell.c -o dogeshell.o
dogec -c kernel/dogescript.c -o dogescript.o
dogec -c kernel/file.c -o file.o
dogec -c kernel/kernel.c -o kernel.o

# link
ld -m elf_i386 -T linker/linker.ld -o windoge.bin boot.o kernel.o data.o info.o serial.o dogescript.o dogeshell.o dogeio.o ports.o string.o suchboot.o time.o vbe.o dogeio_graphics.o font.o file.o

# delete
rm boot.o

rm dogeio.o
rm time.o
rm string.o
rm ports.o
rm vbe.o
rm dogeio_graphics.o
rm font.o
rm file.o

rm suchboot.o
rm data.o
rm info.o
rm serial.o
rm dogeshell.o
rm dogescript.o
rm kernel.o

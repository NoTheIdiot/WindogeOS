# boot/
nasm -f elf32 boot/boot.asm -o boot.o

# helper/
dogec -c helper/dogeio.c -o dogeio.o
dogec -c helper/time.c -o time.o
dogec -c helper/string.c -o string.o
dogec -c helper/ports.c -o ports.o

# kernel/
dogec -c kernel/suchboot.c -o suchboot.o
dogec -c kernel/data.c -o data.o
dogec -c kernel/info.c -o info.o
dogec -c kernel/kernel.c -o kernel.o

# link
ld -m elf_i386 -T linker/linker.ld -o windoge.bin boot.o kernel.o data.o info.o dogeio.o ports.o string.o suchboot.o time.o

# delete
rm boot.o

rm dogeio.o
rm time.o
rm string.o
rm ports.o

rm suchboot.o
rm data.o
rm info.o
rm kernel.o
